ALTER DATABASE externaldb SET datestyle TO "ISO, DMY";

CREATE TABLE courier_actions (
  courier_id INT,
  order_id INT,
  time TIMESTAMP,
  action VARCHAR(255)
);

CREATE TABLE couriers (
  courier_id INT PRIMARY KEY,
  birth_date DATE,
  sex VARCHAR(255)
);

CREATE TABLE orders (
  order_id INT PRIMARY KEY,
  creation_time TIMESTAMP,
  product_ids TEXT
);

CREATE TABLE products (
  product_id INT PRIMARY KEY,
  price NUMERIC,
  name VARCHAR(255)
);

CREATE TABLE users (
  user_id INT PRIMARY KEY,
  birth_date DATE,
  sex VARCHAR(255)
);

CREATE TABLE user_actions (
  user_id INT,
  order_id INT,
  time TIMESTAMP,
  action VARCHAR(255)
);
